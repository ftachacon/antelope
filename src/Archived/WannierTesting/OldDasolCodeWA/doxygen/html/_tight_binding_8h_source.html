76 TightBinding::TightBinding( const libconfig::Setting *params )
77 {
78  try
79  {
80  string w90_file = params->lookup("dataName");
81  ifstream w90_data(w90_file.c_str());
82 
83  int itemp, jtemp;
84  double ftemp1, ftemp2;
85  if (w90_data.is_open())
86  {
87  string line;
88  getline(w90_data, line); // First line is header
89  // lattice_vector
90  for (int i = 0; i < Ndim; ++i)
91  {
92  w90_data >> vec_lattice[i][0] >> vec_lattice[i][1] >> vec_lattice[i][2];
93  }
94  w90_data >> Nband >> Nrpts; // num_wann, nrpts in Wannier90
95  Allocate(); // allocate
96  for (int irpt = 0; irpt < Nrpts; ++irpt)
97  {
98  w90_data >> weight[irpt];
99  weight[irpt] = 1.0/weight[irpt];
100  }
101  // Read Hamiltonian components
102  for (int irpt = 0; irpt < Nrpts; ++irpt)
103  {
104  // Read R vector
105  w90_data >> indexRvec[irpt][0] >> indexRvec[irpt][1] >> indexRvec[irpt][2];
106  // read Hamitonian components corresponding to <0m|H|Rn>
107  for (int m = 0; m < Nband*Nband; ++m)
108  {
109  w90_data >> jtemp >> itemp >> ftemp1 >> ftemp2;
110  ham_w[irpt][jtemp][itemp] = ftemp1 * I*ftemp2;
111  }
112  }
113  // Read position matrix elements
114  for (int irpt = 0; irpt < Nrpts; ++irpt)
115  {
116  // Read R vector
117  // Below is repeated behavior - maybe I can add error-checking routine here
118  w90_data >> indexRvec[irpt][0] >> indexRvec[irpt][1] >> indexRvec[irpt][2];
119  if (indexRvec[irpt][0] == 0 && indexRvec[irpt][1] == 0 && indexRvec[irpt][2] == 0)
120  indexRorigin = irpt;
121  for (int m = 0; m < Nband*Nband; ++m)
122  {
123  w90_data >> jtemp >> itemp;
124  for (int i = 0; i < Ndim; ++i)
125  {
126  w90_data >> ftemp1 >> ftemp2;
127  pos_w[irpt][jtemp][itemp][i] = ftemp1 * I*ftemp2;
128  }
129  }
130  }
131  }
132 
133  // construct r vectors
134  for (int m = 0; m < Nband; ++m)
135  {
136  for (int n = 0; n < Nband; ++n)
137  {
138  for (int irpt = 0; irpt < Nrpts; ++irpt)
139  {
140  for (int iaxis = 0; iaxis < 3; ++iaxis)
141  rvec[irpt][m][n][iaxis] = indexRvec[irpt][0]*vec_lattice[0][iaxis]
142  + indexRvec[irpt][1]*vec_lattice[1][iaxis]
143  + indexRvec[irpt][2]*vec_lattice[2][iaxis]
144  + real(pos_w[irpt][n][n][iaxis] - pos_w[irpt][m][m][iaxis]);
145  }
146  }
147  }
148 
149  // 'S' for safe minimum (1/sfmin != nan)
150  // 'E' for machine epsilon
151  eps = LAPACKE_dlamch('S');
152  }
153  catch(const std::exception& e)
154  {
155  std::cerr << e.what() << '\n';
156  }
157 
158  SetReciprocalBasis();
159 }

287 void TightBinding::SetReciprocalBasis()
288 {
289  Volume = 0.;
290  Volume += vec_lattice[0][0] * (vec_lattice[1][1]*vec_lattice[2][2] - vec_lattice[1][2]*vec_lattice[2][1])
291  + vec_lattice[0][1] * (vec_lattice[1][2]*vec_lattice[2][0] - vec_lattice[1][0]*vec_lattice[2][2])
292  + vec_lattice[0][2] * (vec_lattice[1][0]*vec_lattice[2][1] - vec_lattice[1][1]*vec_lattice[2][0]);
293  vec_reciprocal[0] = CrossProduct(vec_lattice[1], vec_lattice[2]);
294  vec_reciprocal[1] = CrossProduct(vec_lattice[2], vec_lattice[0]);
295  vec_reciprocal[2] = CrossProduct(vec_lattice[0], vec_lattice[1]);
296  for (int i = 0; i < Ndim; ++i) for (int j = 0; j < Ndim; ++j) vec_reciprocal[i][j] /= Volume;
297 
298  BZorigin[0] = 0.; BZorigin[1] = 0.; BZorigin[2] = 0.;
299  for (int i = 0; i < Ndim; ++i) for (int j = 0; j < Ndim; ++j) BZaxis[i*Ndim + j] = vec_reciprocal[i][j];
