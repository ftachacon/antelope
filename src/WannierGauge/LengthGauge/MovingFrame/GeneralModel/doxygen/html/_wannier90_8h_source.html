107 Wannier90::Wannier90( const libconfig::Setting *params )
108 {
109  isMeshAllocated = false;
110  try
111  {
112  string w90_file = params->lookup("dataName");
113  // If FermiE exists, valence band number is ignored
114  isFermiEUsed = params->lookupValue("FermiE", FermiE);
115  if (isFermiEUsed)
116  FermiE = FermiE / au_eV;
117  else
118  Nval = params->lookup("Nval");
119  thermalE = 300.;
120  params->lookupValue("temperature", thermalE);
121  thermalE /= 3.15775024804e5; // hartree energy (4.3597447222071×10−18) / Boltzman constant(1.380649×10−23)
122 
123  isDipoleZero = false;
124  //isDipoleZero = true;
125  //params->lookupValue("isDipoleZero", isDipoleZero);
126  ifstream w90_data(w90_file.c_str());
127 
128  int itemp, jtemp;
129  double ftemp1, ftemp2;
130  if (w90_data.is_open())
131  {
132  string line;
133  getline(w90_data, line); // First line is header
134  // lattice_vector
135  for (int i = 0; i < Ndim; ++i)
136  {
137  w90_data >> vec_lattice[i][0] >> vec_lattice[i][1] >> vec_lattice[i][2];
138  }
139  for (int i = 0; i < Ndim; ++i) for (int j = 0; j < Ndim; ++j) vec_lattice[i][j] /= au_angstrom;
140  w90_data >> Nband >> Nrpts; // num_wann, nrpts in Wannier90
141  Allocate(); // allocate
142  for (int irpt = 0; irpt < Nrpts; ++irpt)
143  {
144  w90_data >> weight[irpt];
145  weight[irpt] = 1.0/weight[irpt];
146  }
147  // Read Hamiltonian components
148  for (int irpt = 0; irpt < Nrpts; ++irpt)
149  {
150  // Read R vector
151  w90_data >> indexRvec[irpt][0] >> indexRvec[irpt][1] >> indexRvec[irpt][2];
152  // read Hamitonian components corresponding to <0m|H|Rn>
153  for (int m = 0; m < Nband*Nband; ++m)
154  {
155  w90_data >> jtemp >> itemp >> ftemp1 >> ftemp2;
156  ham_w[irpt][jtemp-1][itemp-1] = (ftemp1 + I*ftemp2)/au_eV;
157  }
158  }
159  // Read position matrix elements
160  for (int irpt = 0; irpt < Nrpts; ++irpt)
161  {
162  // Read R vector
163  // Below is repeated behavior - maybe I can add error-checking routine here
164  w90_data >> indexRvec[irpt][0] >> indexRvec[irpt][1] >> indexRvec[irpt][2];
165  if (indexRvec[irpt][0] == 0 && indexRvec[irpt][1] == 0 && indexRvec[irpt][2] == 0)
166  indexRorigin = irpt;
167  for (int m = 0; m < Nband*Nband; ++m)
168  {
169  w90_data >> jtemp >> itemp;
170  for (int i = 0; i < Ndim; ++i)
171  {
172  w90_data >> ftemp1 >> ftemp2;
173  pos_w[irpt][jtemp-1][itemp-1][i] = (ftemp1 * I*ftemp2)/au_angstrom;
174  }
175  }
176  }
177  }
178 
179  // construct r vectors
180  #pragma omp parallel for collapse(4)
181  for (int m = 0; m < Nband; ++m)
182  {
183  for (int n = 0; n < Nband; ++n)
184  {
185  for (int irpt = 0; irpt < Nrpts; ++irpt)
186  {
187  for (int iaxis = 0; iaxis < Ndim; ++iaxis)
188  rvec[irpt][m][n][iaxis] = indexRvec[irpt][0]*vec_lattice[0][iaxis]
189  + indexRvec[irpt][1]*vec_lattice[1][iaxis]
190  + indexRvec[irpt][2]*vec_lattice[2][iaxis]
191  + real(pos_w[indexRorigin][n][n][iaxis] - pos_w[indexRorigin][m][m][iaxis]);
192  }
193  }
194  }
195 
196  // construct R vectors
197  #pragma omp parallel for collapse(2)
198  for (int irpt = 0; irpt < Nrpts; ++irpt)
199  {
200  for (int iaxis = 0; iaxis < Ndim; ++iaxis)
201  {
202  Rvec[irpt][iaxis] = indexRvec[irpt][0]*vec_lattice[0][iaxis]
203  + indexRvec[irpt][1]*vec_lattice[1][iaxis]
204  + indexRvec[irpt][2]*vec_lattice[2][iaxis];
205  }
206  }
207 
208  // 'S' for safe minimum (1/sfmin != nan)
209  // 'E' for machine epsilon
210  eps = LAPACKE_dlamch('S');
211  }
212  catch(const std::exception& e)
213  {
214  std::cerr << e.what() << '\n';
215  }
216 
217  SetReciprocalBasis();
218 }

265 void Wannier90::GenInitialValue(complex *_dmstore, std::array<double, Ndim> _kpoint)
266 {
267  int ithread = omp_get_thread_num();
268  fill(_dmstore, _dmstore + Nband*Nband, 0.);
269  GenUMatrix(tempUmat[ithread], _kpoint);
270  for (int m = 0; m < Nband; ++m)
271  {
272  for (int n = 0; n < Nband; ++n)
273  {
274  tempctransUmat[ithread][m*Nband + n] = conj( tempUmat[ithread][n*Nband + m] );
275  }
276  }
277 
278  // set valence = 1., conduction = 0. initial condition
279  // Fermi-Dirac distribution is used when FermiE is applied
280  // tempEigval is calculated inside the GenUMatrix, be careful about order or sideeffects.
281  for (int m = 0; m < Nband; ++m)
282  {
283  if (isFermiEUsed)
284  {
285  _dmstore[m*Nband + m] = 1.0 / ( exp( (tempEigval[ithread][m] - FermiE) / thermalE ) + 1.0 );
286  }
287  else
288  {
289  if (m < Nval)
290  _dmstore[m*Nband + m] = 1.;
291  }
292  }
293  MatrixMult(tempMatrix[ithread], _dmstore, tempctransUmat[ithread], Nband);
294  MatrixMult(_dmstore, tempUmat[ithread], tempMatrix[ithread], Nband);
295 }

508 void Wannier90::SetReciprocalBasis()
509 {
510  Volume = 0.;
511  Volume += vec_lattice[0][0] * (vec_lattice[1][1]*vec_lattice[2][2] - vec_lattice[1][2]*vec_lattice[2][1])
512  + vec_lattice[0][1] * (vec_lattice[1][2]*vec_lattice[2][0] - vec_lattice[1][0]*vec_lattice[2][2])
513  + vec_lattice[0][2] * (vec_lattice[1][0]*vec_lattice[2][1] - vec_lattice[1][1]*vec_lattice[2][0]);
514  vec_reciprocal[0] = CrossProduct(vec_lattice[1], vec_lattice[2]);
515  vec_reciprocal[1] = CrossProduct(vec_lattice[2], vec_lattice[0]);
516  vec_reciprocal[2] = CrossProduct(vec_lattice[0], vec_lattice[1]);
517  for (int i = 0; i < Ndim; ++i) for (int j = 0; j < Ndim; ++j) vec_reciprocal[i][j] *= (2*pi)/Volume;
518 
519  BZorigin[0] = 0.; BZorigin[1] = 0.; BZorigin[2] = 0.;
520  for (int i = 0; i < Ndim; ++i) for (int j = 0; j < Ndim; ++j) BZaxis[i*Ndim + j] = vec_reciprocal[i][j];
521 }
