67 void WilsonMass::GenInitialValue(complex *_dmstore, std::array<double, Ndim> _kpoint)
68 {
69  GenBcomp(_kpoint);
70 
71  double Bnorm = sqrt( Bcomp[1]*Bcomp[1] + Bcomp[2]*Bcomp[2] + Bcomp[3]*Bcomp[3] );
72  if (Bnorm > eps)
73  {
74  _dmstore[0] = 0.5 - Bcomp[3]/(2.*Bnorm); _dmstore[1] = -(Bcomp[1] - I*Bcomp[2])/(2.*Bnorm);
75  _dmstore[2] = conj(_dmstore[1]); _dmstore[3] = 1.0 - _dmstore[0];
76  }
77  else
78  {
79  // B1, B2 --> 0, B3 < 0 case
80  _dmstore[0] = 0.5 - Bcomp[3]/(2.*Bnorm); _dmstore[1] = -(Bcomp[1] - I*Bcomp[2])/(2.*Bnorm);
81  _dmstore[2] = conj(_dmstore[1]); _dmstore[3] = 1.0 - _dmstore[0];
82  }

84 }

86 void WilsonMass::GenUMatrix(complex *_ustore, std::array<double, Ndim> _kpoint)
87 {
88  GenBcomp(_kpoint);
89  double Bnorm = sqrt( Bcomp[1]*Bcomp[1] + Bcomp[2]*Bcomp[2] + Bcomp[3]*Bcomp[3] );
90  double waveNorm = sqrt( 2.*Bnorm* (Bnorm + Bcomp[3]) );
91  if (waveNorm > eps)
92  {
93  _ustore[0] = (Bcomp[3] + Bnorm)/waveNorm; _ustore[1] = -(Bcomp[1] - I*Bcomp[2])/waveNorm;
94  _ustore[2] = (Bcomp[1] + I*Bcomp[2])/waveNorm; _ustore[3] = (Bcomp[3] + Bnorm)/waveNorm;
95  }
96  else
97  {
98  // B1, B2 --> 0, B3 < 0 case
99  _ustore[0] = 0.; _ustore[1] = 1.;
100  _ustore[2] = 1.; _ustore[3] = 0.;
101  }
102 }

112 void WilsonMass::GenJMatrix(complex **_jstore, std::array<double, Ndim> _kpoint)
113 {
114  double xderBcomp[4]; double yderBcomp[4];
115 
116  Bcomp[0] = 0.;
117  Bcomp[1] = delta*sin(a0*_kpoint[0]); Bcomp[2] = delta*sin(a1*_kpoint[1]);
118  Bcomp[3] = -2.*t*(cos(a0*_kpoint[0]) + cos(a1*_kpoint[1])) - mu;
119 
120  xderBcomp[0] = 0.; yderBcomp[0] = 0.;
121  xderBcomp[1] = a0*delta*cos(a0*_kpoint[0]); yderBcomp[1] = 0.;
122  xderBcomp[2] = 0.; yderBcomp[2] = a1*delta*cos(a1*_kpoint[1]);
123  xderBcomp[3] = 2.*t *a0* sin(a0*_kpoint[0]); yderBcomp[3] = 2.*t *a1* sin(a1*_kpoint[1]);
124 
125  _jstore[0][0] = -(xderBcomp[0] + xderBcomp[3]); _jstore[1][0] = -(yderBcomp[0] + yderBcomp[3]);
126  _jstore[0][1] = -(xderBcomp[1] - I*xderBcomp[2]); _jstore[1][1] = -(yderBcomp[1] - I*yderBcomp[2]);
127  _jstore[0][2] = -(xderBcomp[1] + I*xderBcomp[2]); _jstore[1][2] = -(yderBcomp[1] + I*yderBcomp[2]);
128  _jstore[0][3] = -(xderBcomp[0] - xderBcomp[3]); _jstore[1][3] = -(yderBcomp[0] - yderBcomp[3]);
