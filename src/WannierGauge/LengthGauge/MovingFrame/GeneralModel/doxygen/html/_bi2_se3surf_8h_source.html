235 void BieSe3surf::GenBcomp(std::array<double, Ndim> _kpoint)
236 {
237  fill(Bcomp, Bcomp+4, 0.);
238  double cosAvecSum = 0.;
239  double sinAvecSum = 0.;
240  double cosBvecSum = 0.;
241  double sinBvecSum = 0.;
242  for (int i = 0; i < 3; ++i)
243  {
244  angle_a0[i] = _kpoint[0]*vec_a[i][0] + _kpoint[1]*vec_a[i][1];
245  angle_b0[i] = _kpoint[0]*vec_b[i][0] + _kpoint[1]*vec_b[i][1];
246 
247  cosAvecSum += cos(angle_a0[i]);
248  sinAvecSum += sin(angle_a0[i]);
249  cosBvecSum += cos(angle_b0[i]);
250  sinBvecSum += sin(angle_b0[i]);
251  }
252 
253  double w = -2.*pi/3.;
254 
255  hcomp[0] = 2.*acomp[0] * cosAvecSum + 2.*bcomp[0] * cosBvecSum;
256 
257  hcomp[1] = -2.*acomp[3] * sin( w)*( sin(angle_a0[1]) - sin(angle_a0[2]) )
258  + 2.*bcomp[3] * ( sin(angle_b0[0]) + cos(w)*(sin(angle_b0[1]) - sin(angle_b0[2]) ) ) ;
259 
260  hcomp[2] = -2.*bcomp[3] * sin( w )*( sin(angle_b0[1]) - sin(angle_b0[2]) )
261  -2.*acomp[3] * ( sin(angle_a0[0]) + cos( w )*(sin(angle_a0[1]) + sin(angle_a0[2]) ) );
262 
263  hcomp[3] = 2.*acomp[2] * sinAvecSum;
264 
265  hcomp[4] = -2.*bcomp[2] * sinBvecSum;
266 
267  hcomp[5] = 2.*acomp[1] * cosAvecSum + 2.*bcomp[1] * cosBvecSum + m11;
268 
269  double hcomp5gamma = 6.*acomp[1] + 6.*bcomp[1] + m11;
270  double bcoeff = sqrt(1. - bcomp[0]*bcomp[0]/(bcomp[1]*bcomp[1]));
271 
272  Bcomp[0] = hcomp[0] + bcomp[0]/bcomp[1] * (-hcomp[5] + hcomp5gamma);
273 
274  Bcomp[1] = bcoeff * hcomp[1];
275 
276  Bcomp[2] = bcoeff * hcomp[2];
277 
278  Bcomp[3] = bcoeff * hcomp[3];
279 }
